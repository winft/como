add_library(utils INTERFACE)

target_sources(utils
  PUBLIC
    FILE_SET HEADERS
    FILES
      algorithm.h
      blocker.h
      file_descriptor.h
      flags.h
      gamma_ramp.h

      memory.h

  EXPORT como-export
  LIBRARY NAMELINK_SKIP
  FILE_SET HEADERS DESTINATION ${KDE_INSTALL_INCLUDEDIR}/como/utils
)
